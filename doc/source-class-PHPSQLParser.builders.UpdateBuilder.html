61:     public function build(array $parsed) {
62:         $sql = '';

64:         foreach ($parsed as $k => $v) {
65:             $len = strlen($sql);
66:             $sql .= $this->buildTable($v, $k);

69:                 throw new UnableToCreateSQLException('UPDATE table list', $k, $v, 'expr_type');
