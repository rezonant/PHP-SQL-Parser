 46:  * This class processes the UNION statements.
 47:  *
 48:  * @author  André Rothe <andre.rothe@phosco.info>
 49:  * @license http://www.debian.org/misc/bsd.license  BSD License (3 Clause)
 50:  *

 52: class UnionProcessor extends AbstractProcessor {
 53: 
 54:     protected function processDefault($token) {
 55:         $processor = new DefaultProcessor();
 56:         return $processor->process($token);
 57:     }
 58: 
 59:     protected function processSQL($token) {
 60:         $processor = new SQLProcessor();
 61:         return $processor->process($token);
 62:     }
 63: 
 64:     public static function isUnion($queries) {
 65:         $unionTypes = array('UNION', 'UNION ALL');
 66:         foreach ($unionTypes as $unionType) {
 67:             if (!empty($queries[$unionType])) {
 68:                 return true;
 69:             }
 70:         }
 71:         return false;
 72:     }
 73: 
 74:     /**
 75:      * MySQL supports a special form of UNION:
 76:      * (select ...)
 77:      * union
 78:      * (select ...)
 79:      *
 80:      * This function handles this query syntax. Only one such subquery
 81:      * is supported in each UNION block. (select)(select)union(select) is not legal.
 82:      * The extra queries will be silently ignored.
 83:      */
 84:     protected function processMySQLUnion($queries) {
 85:         $unionTypes = array('UNION', 'UNION ALL');
 86:         foreach ($unionTypes as $unionType) {
 87: 
 88:             if (empty($queries[$unionType])) {
 89:                 continue;
 90:             }
 91: 
 92:             foreach ($queries[$unionType] as $key => $tokenList) {
 93:                 foreach ($tokenList as $z => $token) {
 94:                     $token = trim($token);
 95:                     if ($token === "") {
 96:                         continue;
 97:                     }
 98: 
 99:                     // starts with "(select"
100:                     if (preg_match("/^\\(\\s*select\\s*/i", $token)) {
101:                         $queries[$unionType][$key] = $this->processDefault($this->removeParenthesisFromStart($token));
102:                         break;
103:                     }
104: 
105:                     $queries[$unionType][$key] = $this->processSQL($queries[$unionType][$key]);
106:                     break;
107:                 }
108:             }
109:         }
110:         // it can be parsed or not
111:         return $queries;
112:     }
113: 
114:     public function process($inputArray) {
115:         $outputArray = array();
116: 
117:         // ometimes the parser needs to skip ahead until a particular
118:         // oken is found
119:         $skipUntilToken = false;
120: 
121:         // his is the last type of union used (UNION or UNION ALL)
122:         // ndicates a) presence of at least one union in this query
123:         // b) the type of union if this is the first or last query
124:         $unionType = false;
125: 
126:         // ometimes a "query" consists of more than one query (like a UNION query)
127:         // his array holds all the queries
128:         $queries = array();
129: 
130:         foreach ($inputArray as $key => $token) {
131:             $trim = trim($token);
132: 
133:             // overread all tokens till that given token
134:             if ($skipUntilToken) {
135:                 if ($trim === "") {
136:                     continue; // read the next token
137:                 }
138:                 if (strtoupper($trim) === $skipUntilToken) {
139:                     $skipUntilToken = false;
140:                     continue; // read the next token
141:                 }
142:             }
143: 
144:             if (strtoupper($trim) !== "UNION") {
145:                 $outputArray[] = $token; // here we get empty tokens, if we remove these, we get problems in parse_sql()
146:                 continue;
147:             }
148: 
149:             $unionType = "UNION";
150: 
151:             // we are looking for an ALL token right after UNION
152:             for ($i = $key + 1; $i < count($inputArray); ++$i) {
153:                 if (trim($inputArray[$i]) === "") {
154:                     continue;
155:                 }
156:                 if (strtoupper($inputArray[$i]) !== "ALL") {
157:                     break;
158:                 }
159:                 // the other for-loop should overread till "ALL"
160:                 $skipUntilToken = "ALL";
161:                 $unionType = "UNION ALL";
162:             }
163: 
164:             // store the tokens related to the unionType
165:             $queries[$unionType][] = $outputArray;
166:             $outputArray = array();
167:         }
168: 
169:         // the query tokens after the last UNION or UNION ALL
170:         // or we don't have an UNION/UNION ALL
171:         if (!empty($outputArray)) {
172:             if ($unionType) {
173:                 $queries[$unionType][] = $outputArray;
174:             } else {
175:                 $queries[] = $outputArray;
176:             }
177:         }
178: 
179:         return $this->processMySQLUnion($queries);
180:     }
