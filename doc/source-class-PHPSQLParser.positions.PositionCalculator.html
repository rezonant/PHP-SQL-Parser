129:     protected function findPositionWithinString($sql, $value, $expr_type) {
130: 
131:         $offset = 0;
132:         $ok = false;
133:         while (true) {
134: 
135:             $pos = strpos($sql, $value, $offset);
136:             if ($pos === false) {
137:                 break;
138:             }
139: 
140:             $before = "";
141:             if ($pos > 0) {
142:                 $before = $sql[$pos - 1];
143:             }
144: 
145:             // if we have a quoted string, we every character is allowed after it
146:             // see issue 137
147:             $quoted = ($sql[$pos + strlen($value) - 1] === '`');
148:             $after = "";
149:             if (isset($sql[$pos + strlen($value)])) {
150:                 $after = $sql[$pos + strlen($value)];
151:             }
152: 
153:             // if we have an operator, it should be surrounded by
154:             // whitespace, comma, parenthesis, digit or letter, end_of_string
155:             // an operator should not be surrounded by another operator
156: 
157:             if ($expr_type === 'operator') {
158: 
159:                 $ok = ($before === "" || in_array($before, self::$allowedOnOperator, true))
160:                     || (strtolower($before) >= 'a' && strtolower($before) <= 'z');
161:                 $ok = $ok
162:                     && ($after === "" || in_array($after, self::$allowedOnOperator, true)
163:                         || (strtolower($after) >= 'a' && strtolower($after) <= 'z'));
164: 
165:                 if (!$ok) {
166:                     $offset = $pos + 1;
167:                     continue;
168:                 }
169: 
170:                 break;
171:             }
172: 
173:             // in all other cases we accept
174:             // whitespace, comma, operators, parenthesis and end_of_string
175: 
176:             $ok = ($before === "" || in_array($before, self::$allowedOnOther, true));
177:             $ok = $ok
178:                 && ($after === "" || in_array($after, self::$allowedOnOther, true)
179:                     || ($quoted && (strtolower($after) >= 'a' && strtolower($after) <= 'z')));
180: 
181:             if ($ok) {
182:                 break;
183:             }
184: 
185:             $offset = $pos + 1;
186:         }
187: 
188:         return $pos;
189:     }

191:     protected function lookForBaseExpression($sql, &$charPos, &$parsed, $key, &$backtracking) {
192:         if (!is_numeric($key)) {
193:             if (($key === 'UNION' || $key === 'UNION ALL')
194:                 || ($key === 'expr_type' && isset($this->flippedBacktrackingTypes[$parsed]))
195:                 || ($key === 'select-option' && $parsed !== false) || ($key === 'alias' && $parsed !== false)) {
196:                 // we hold the current position and come back after the next base_expr
197:                 // we do this, because the next base_expr contains the complete expression/subquery/record
198:                 // and we have to look into it too
199:                 $backtracking[] = $charPos;
200: 
201:             } elseif (($key === 'ref_clause' || $key === 'columns') && $parsed !== false) {
202:                 // we hold the current position and come back after n base_expr(s)
203:                 // there is an array of sub-elements before (!) the base_expr clause of the current element
204:                 // so we go through the sub-elements and must come at the end
205:                 $backtracking[] = $charPos;
206:                 for ($i = 1; $i < count($parsed); $i++) {
207:                     $backtracking[] = false; // backtracking only after n base_expr!
208:                 }
209:             } elseif (($key === 'sub_tree' && $parsed !== false) || ($key === 'options' && $parsed !== false)) {
210:                 // we prevent wrong backtracking on subtrees (too much array_pop())
211:                 // there is an array of sub-elements after(!) the base_expr clause of the current element
212:                 // so we go through the sub-elements and must not come back at the end
213:                 for ($i = 1; $i < count($parsed); $i++) {
214:                     $backtracking[] = false;
215:                 }
216:             } elseif (($key === 'TABLE') || ($key === 'create-def' && $parsed !== false)) {
217:                 // do nothing
218:             } else {
219:                 // move the current pos after the keyword
220:                 // SELECT, WHERE, INSERT etc.
221:                 if (PHPSQLParserConstants::getInstance()->isReserved($key)) {
222:                     $charPos = stripos($sql, $key, $charPos);
223:                     $charPos += strlen($key);
224:                 }
225:             }
226:         }
227: 
228:         if (!is_array($parsed)) {
229:             return;
230:         }
231: 
232:         foreach ($parsed as $key => $value) {
233:             if ($key === 'base_expr') {
234: 
235:                 //$this->printPos("0", $sql, $charPos, $key, $value, $backtracking);
236: 
237:                 $subject = substr($sql, $charPos);
238:                 $pos = $this->findPositionWithinString($subject, $value,
239:                     isset($parsed['expr_type']) ? $parsed['expr_type'] : 'alias');
240:                 if ($pos === false) {
241:                     throw new UnableToCalculatePositionException($value, $subject);
242:                 }
243: 
244:                 $parsed['position'] = $charPos + $pos;
245:                 $charPos += $pos + strlen($value);
246: 
247:                 //$this->printPos("1", $sql, $charPos, $key, $value, $backtracking);
248: 
249:                 $oldPos = array_pop($backtracking);
250:                 if (isset($oldPos) && $oldPos !== false) {
251:                     $charPos = $oldPos;
252:                 }
253: 
254:                 //$this->printPos("2", $sql, $charPos, $key, $value, $backtracking);
255: 
256:             } else {
257:                 $this->lookForBaseExpression($sql, $charPos, $parsed[$key], $key, $backtracking);
258:             }
259:         }
260:     }
