51: class LimitProcessor extends AbstractProcessor {

53:     public function process($tokens) {
54:         $rowcount = "";
55:         $offset = "";
56: 
57:         $comma = -1;
58:         $exchange = false;
59: 
60:         for ($i = 0; $i < count($tokens); ++$i) {
61:             $trim = strtoupper(trim($tokens[$i]));
62:             if ($trim === ",") {
63:                 $comma = $i;
64:                 break;
65:             }
66:             if ($trim === "OFFSET") {
67:                 $comma = $i;
68:                 $exchange = true;
69:                 break;
70:             }
71:         }
72: 
73:         for ($i = 0; $i < $comma; ++$i) {
74:             if ($exchange) {
75:                 $rowcount .= $tokens[$i];
76:             } else {
77:                 $offset .= $tokens[$i];
78:             }
79:         }
80: 
81:         for ($i = $comma + 1; $i < count($tokens); ++$i) {
82:             if ($exchange) {
83:                 $offset .= $tokens[$i];
84:             } else {
85:                 $rowcount .= $tokens[$i];
86:             }
87:         }
88: 
89:         return array('offset' => trim($offset), 'rowcount' => trim($rowcount));
90:     }
91: }
