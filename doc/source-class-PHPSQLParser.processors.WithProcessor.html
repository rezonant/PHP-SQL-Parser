 33: namespace PHPSQLParser\processors;
 34: use PHPSQLParser\utils\ExpressionType;
 35: 
 36: /**
 37:  * 
 38:  * This class processes Oracle's WITH statements.
 39:  * 
 40:  * @author  André Rothe <andre.rothe@phosco.info>
 41:  * @license http://www.debian.org/misc/bsd.license  BSD License (3 Clause)
 42:  * 
 43:  */
 44: class WithProcessor extends AbstractProcessor {
 45: 
 46:     protected function processTopLevel($sql) {
 47:         $processor = new DefaultProcessor();
 48:         return $processor->process($sql);
 49:     }
 50:     
 51:     protected function buildTableName($token) {
 52:         return array('expr_type' => ExpressionType::TEMPORARY_TABLE, 'name'=>$token, 'base_expr' => $token, 'no_quotes' => $this->revokeQuotation($token));
 53:     }
 54:     
 55:     public function process($tokens) {
 56:         $out = array();
 57:         $resultList = array();
 58:         $category = '';
 59:         $base_expr = '';
 60:         $prev = '';
 61: 
 62:         foreach ($tokens as $token) {
 63:             $base_expr .= $token;
 64:             $upper = strtoupper(trim($token));
 65:             
 66:             if ($this->isWhitespaceToken($token)) {
 67:                 continue;
 68:             }
 69: 
 70:             $trim = trim($token);
 71:             switch ($upper) {
 72: 
 73:             case 'AS':
 74:                 if ($prev !== 'TABLENAME') {
 75:                     // error or tablename is AS
 76:                     $resultList[] = $this->buildTableName($trim);
 77:                     $category = 'TABLENAME';
 78:                     break;
 79:                 }
 80:                 
 81:                 $resultList[] = array('expr_type' => ExpressionType::RESERVED, 'base_expr' => $trim);
 82:                 $category = $upper;
 83:                 break;
 84: 
 85:             case ',':
 86:                 // ignore
 87:                 $base_expr = '';
 88:                 break;    
 89:                 
 90:             default:
 91:                 switch ($prev) {
 92:                     case 'AS':
 93:                         // it follows a parentheses pair    
 94:                         $subtree = $this->processTopLevel($this->removeParenthesisFromStart($token));
 95:                         $resultList[] = array('expr_type' => ExpressionType::BRACKET_EXPRESSION, 'base_expr' => $trim, 'sub_tree' => $subtree);
 96:                         
 97:                         $out[] = array('expr_type' => ExpressionType::SUBQUERY_FACTORING, 'base_expr' => trim($base_expr), 'sub_tree' => $resultList);
 98:                         $resultList = array();
 99:                         $category = '';
100:                     break;
101:                     
102:                     case '':
103:                         // we have the name of the table
104:                         $resultList[] = $this->buildTableName($trim);
105:                         $category = 'TABLENAME';
106:                         break;
107:                         
108:                 default:
109:                 // ignore
110:                     break;
111:                 }
112:                 break;
113:             }
114:             $prev = $category;
115:         }
116:         return $out;
117:     }
118: }
